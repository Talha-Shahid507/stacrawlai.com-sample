example.py
from stackcrawlai import Crawler

# Initialize with your API key
crawler = Crawler(api_key="sk-...")

# Describe what you want in plain English
result = crawler.extract(
  url="https://shop.example.com",
  prompt="Get all product names, prices, ratings",
  output="json",
  depth=3
)

# Clean structured data, instantly
print(result.data) # 847 products extracted ✓
